function subcribeNewletter(){
	if($('#username').val()!=''){
		saveSubcribeNewletter($('#fullname').val(),$('#username').val());
	}else{
		$('#sn_fullname').val('Họ tên');
		$('#sn_email').val('Email');
		showDialogbox('subscription_popup');
	}
}
function checkFormSubcribe(){
	var has_error = false;
    $('#err_sn_fullname').css('display','none');
    if($('#sn_fullname').val() == "Họ tên"){
        $('#err_sn_fullname').css('display','');has_error=true;
    }else{
        if(!checkRequired(frm_sn.sn_fullname))has_error = true;
    }
	if(!checkEmail(frm_sn.sn_email))has_error = true;
	if (!has_error) {
		saveSubcribeNewletter($('#sn_fullname').val(),$('#sn_email').val());
	}else{
		return false;
	}
}
function saveSubcribeNewletter(fullname,email){
	$.post("/jobseekers/save_visitor_newsletter.php",{
		sn_fullname : fullname,
		sn_email    : email
	}, function(data) {
        closeBlockUI();
        if(data==1){
            if(blockUIClosing){
                setTimeout("showMessage('thanks_is_member','',0,710,1,1,1,'','absolute');",500);
            }else{
                showMessage('thanks_is_member','',0,710,1,1,1,'','absolute');
            }
        }else if(data==2 || data==3){
            if(blockUIClosing){
                setTimeout("showMessage('thanks_non_member','',0,710,1,1,1,'','absolute');",500);
            }else{
                showMessage('thanks_non_member','',0,710,1,1,1,'','absolute');
            }
        }
	});
}
function setDefaultValue(obj,value,case_title){
    if(obj.value==""){
        obj.value = value;
        $("#"+obj.name).addClass("hintText");
    }else{
        if(obj.value == value){
            $("#"+obj.name).addClass("hintText");
        }
    }
}
function formatText(obj,value){
    if(obj.value == value){
        obj.value="";
    }
    $("#"+obj.name).removeClass("hintText");
}
